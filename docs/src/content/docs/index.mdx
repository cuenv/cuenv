## Preview

Define your environment and tasks in `env.cue`:

```cue
package cuenv

import "github.com/cuenv/cuenv/schema"

schema.#Cuenv

// Typed configuration
env: {
    NODE_ENV: "development" | "production"
    PORT:     3000
}

// Task definitions
tasks: {
    build: {
        command:   "bun"
        args:      ["run", "build"]
        dependsOn: ["install"]
    }
}
```
